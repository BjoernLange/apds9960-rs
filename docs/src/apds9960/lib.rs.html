1//! This is a platform agnostic Rust driver for the APDS9960 digital proximity, ambient light, RGB
2//! and gesture sensor, based on the [`embedded-hal`] traits.
3//!
4//! [`embedded-hal`]: https://github.com/rust-embedded/embedded-hal
5//!
6//! This driver allows you to:
7//! - Enable/disable the sensor. See: [`enable()`].
8//! - Enable/disable delay between proximity and / or color / ambient light cycles. See: [`enable_wait()`].
9//! - Enable/disable long delay between proximity and / or color / ambient light cycles. See: [`enable_wait_long()`].
10//! - Set the waiting time between proximity and / or color / ambient light cycles. See: [`set_wait_time()`].
11//! - Force an interrupt. See: [`force_interrupt()`].
12//! - Clear all non-gesture interrupts. See: [`clear_interrupts()`].
13//! - Read the device ID. See: [`read_device_id()`].
14//! - Proximity:
15//!     - Enable/disable the proximity sensor. See: [`enable_proximity()`].
16//!     - Enable/disable proximity interrupt generation. See: [`enable_proximity_interrupts()`].
17//!     - Enable/disable proximity saturation interrupt generation. See: [`enable_proximity_saturation_interrupts()`].
18//!     - Read the proximity data. See: [`read_proximity()`].
19//!     - Check whether the proximity data is valid. See: [`is_proximity_data_valid()`].
20//!     - Set the proximity interrupt low/high thresholds. See: [`set_proximity_low_threshold()`].
21//!     - Set the proximity offsets. See: [`set_proximity_offsets()`].
22//!     - Clear proximity interrupt. See: [`clear_proximity_interrupt()`].
23//! - Color / ambient light:
24//!     - Enable/disable the color / ambient light sensor. See: [`enable_light()`].
25//!     - Enable/disable ambient light interrupt generation. See: [`enable_light_interrupts()`].
26//!     - Enable/disable ambient light saturation interrupt generation. See: [`enable_light_saturation_interrupts()`].
27//!     - Check whether the color / ambient light data is valid. See: [`is_light_data_valid()`].
28//!     - Read the color / ambient light data. See: [`read_light()`].
29//!     - Set the color / ambient light integration time. See: [`set_light_integration_time()`].
30//!     - Set the clear light channel interrupt low/high thresholds. See: [`set_light_low_threshold()`].
31//!     - Clear ambient light interrupt. See: [`clear_light_interrupt()`].
32//! - Gesture detection:
33//!     - Enable/disable gesture detection. See: [`enable_gesture()`].
34//!     - Enable/disable gesture mode. See: [`enable_gesture_mode()`].
35//!     - Enable/disable gesture interrupts. See: [`enable_gesture_interrupts()`].
36//!     - Read whether there is valid gesture data available. See: [`is_gesture_data_valid()`].
37//!     - Read the amount of gesture data available. See: [`read_gesture_data_level()`].
38//!     - Set the threshold of amount of available gesture data. See: [`set_gesture_data_level_threshold()`].
39//!     - Read whether the gesture data has overflown. See: [`has_gesture_data_overflown()`].
40//!     - Read the gesture data. See: [`read_gesture_data()`].
41//!     - Set the gesture proximity entry/exit thresholds. See: [`set_gesture_proximity_entry_threshold()`].
42//!     - Set the gesture offsets. See: [`set_gesture_offsets()`].

81//! ## The device
82//!
83//! The APDS-9960 device features advanced gesture detection, proximity detection, digital ambient
84//! light sense (ALS) and color sense (RGBC).
85//!
86//! The communication is done through an I2C bidirectional bus.
87//!
88//! Datasheet:
89//! - [APDS9960](https://docs.broadcom.com/docs/AV02-4191EN)
90//!
91//! ## Usage examples (see also examples folder)
92//!
93//! To use this driver, import this crate and an `embedded_hal` implementation,
94//! then instantiate the device.
95//!
96//! Please find additional examples in this repository: [apds9960-examples]
97//!
98//! [apds9960-examples]: https://github.com/eldruin/apds9960-examples
99//!

317trait BitFlags<T = Self> {
318    const ADDRESS: u8;
319    fn create(value: u8) -> T;
320    fn with(&self, mask: u8, value: bool) -> T {
321        if value {
322            Self::create(self.value() | mask)
323        } else {
324            Self::create(self.value() & !mask)
325        }
326    }
327
328    fn is(&self, mask: u8, value: bool) -> bool {
329        ((self.value() & mask) != 0) == value
330    }
331
332    fn value(&self) -> u8;
333}

438impl<I2C, M: Mode> Apds9960<I2C, M> {
439    /// Create new instance of the APDS9960 device.
440    pub fn new(i2c: I2C) -> Self {
441        Apds9960 {
442            i2c,

444            config1: register::Config1::default(),
445            config2: register::Config2::default(),
446            gconfig1: register::GConfig1::default(),
