7/// Gesture data reading.
8#[maybe_async_cfg::maybe(
9    sync(feature = "nb", keep_self),
10    async(
11        feature = "async",
12        idents(
13            I2c(async = "I2cAsync"),
14            SyncNonBlocking(async = "Async"),

19impl<I2C, E> Apds9960<I2C, SyncNonBlocking>
20where
21    I2C: I2c<Error = E>,
22{
23    /// Read the amount of available data in the gesture FIFO registers.
24    pub async fn read_gesture_data_level(&mut self) -> Result<u8, Error<E>> {
25        self.read_register(Register::GFLVL).await
26    }
27
28    /// Read whether there is valid gesture data available.
29    #[allow(clippy::wrong_self_convention)]
30    pub async fn is_gesture_data_valid(&mut self) -> Result<bool, Error<E>> {
31        let status = self.read_register(Register::GSTATUS).await?;
32        Ok(GStatus::create(status).is(GStatus::GVALID, true))
33    }
34
35    /// Read whether the gesture data has overflown.
36    #[allow(clippy::wrong_self_convention)]
37    pub async fn has_gesture_data_overflown(&mut self) -> Result<bool, Error<E>> {
38        let status = self.read_register(Register::GSTATUS).await?;
39        Ok(GStatus::create(status).is(GStatus::GFOV, true))
40    }
41
42    /// Read gesture data.
43    ///
44    /// Will read the gesture data up to the minimum of: gesture data level, array size.
45    /// Make sure to provide an array with at least the number of elements returned by the
46    /// `read_gesture_data_level()` method multiplied by 4.
47    ///
48    /// The data contents will be organized as follows:
49    /// `[up_dataset0, down_dataset0, left_dataset0, right_dataset0,
50    ///   up_dataset1, down_dataset1, left_dataset1, right_dataset1, ...]`
51    ///
52    /// Blocks as long as not enough data is available.
53    #[maybe_async_cfg::only_if(sync)]
54    pub fn read_gesture_data(&mut self, data: &mut [u8]) -> nb::Result<(), Error<E>> {
55        if !self.is_gesture_data_valid().map_err(nb::Error::Other)? {
56            return Err(nb::Error::WouldBlock);
57        }
58        let level = self.read_gesture_data_level().map_err(nb::Error::Other)?;
59        let byte_count = core::cmp::min(data.len(), 4 * level as usize);
60        self.read_data(Register::GFIFO_U, &mut data[..byte_count])
61            .map_err(nb::Error::Other)?;
62        Ok(())
63    }
64    /// Read gesture data.
65    ///
66    /// Will read the gesture data up to the minimum of: gesture data level, array size.
67    /// Make sure to provide an array with at least the number of elements returned by the
68    /// `read_gesture_data_level()` method multiplied by 4.
69    ///
70    /// The data contents will be organized as follows:
71    /// `[up_dataset0, down_dataset0, left_dataset0, right_dataset0,
72    ///   up_dataset1, down_dataset1, left_dataset1, right_dataset1, ...]`
73    ///
74    /// Blocks as long as not enough data is available.
75    #[maybe_async_cfg::only_if(async)]
76    pub async fn read_gesture_data(&mut self, data: &mut [u8]) -> Result<(), Error<E>> {
77        while !self.is_gesture_data_valid().await? {}
78        let level = self.read_gesture_data_level().await?;
79        let byte_count = core::cmp::min(data.len(), 4 * level as usize);
80        self.read_data(Register::GFIFO_U, &mut data[..byte_count])
81            .await?;
82        Ok(())
83    }
